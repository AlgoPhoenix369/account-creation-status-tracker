class CachedDataSet(DataSet):
    """
    A dataset backed by Ray object references stored in memory.

    This dataset stores references to computed data in Ray's distributed object store,
    allowing cached data to be reused across multiple operations without recomputation.
    """

    __slots__ = ("_object_refs", "_cached_table", "_cache_id")

    def __init__(self, object_refs: list, cache_id: str) -> None:
        """
        Create a CachedDataSet from Ray object references.

        Parameters
        ----------
        object_refs
            List of Ray ObjectRefs pointing to DataSet objects in the object store.
        cache_id
            Unique identifier for this cache entry.
        """
        super().__init__([])
        self._object_refs = object_refs
        self._cached_table: Optional[arrow.Table] = None
        self._cache_id = cache_id

    def __str__(self) -> str:
        return f"CachedDataSet(cache_id={self._cache_id}, num_partitions={len(self._object_refs)})"

    __repr__ = __str__

    @property
    def object_refs(self) -> list:
        """Ray object references to the cached data."""
        return self._object_refs

    @property
    def cache_id(self) -> str:
        """Unique identifier for this cache entry."""
        return self._cache_id

    def _materialize(self) -> arrow.Table:
        """Materialize the cached data from Ray object store into a single Arrow table."""
        if self._cached_table is None:
            import ray
            datasets = ray.get(self._object_refs)
            tables = [ds.to_arrow_table() for ds in datasets if ds.num_rows > 0]
            if tables:
                self._cached_table = arrow.concat_tables(tables)
            else:
                self._cached_table = arrow.table({})
        return self._cached_table

    @property
    def num_rows(self) -> int:
        return self._materialize().num_rows

    @property
    def empty(self) -> bool:
        return len(self._object_refs) == 0

    def to_arrow_table(
        self,
        max_workers: int = 16,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> arrow.Table:
        return self._materialize()

    def to_batch_reader(
        self,
        batch_size: int = DEFAULT_BATCH_SIZE,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> arrow.RecordBatchReader:
        return self._materialize().to_reader(batch_size)

    def sql_query_fragment(
        self,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> str:
        """
        Register the cached table with DuckDB and return a query fragment.

        This follows the pattern used by ArrowTableDataSet.sql_query_fragment().
        """
        table = self._materialize()
        name = f"cached_table_{self._cache_id}"
        if conn is not None:
            conn.register(name, table)
        else:
            duckdb.register(name, table)
        return f"( select * from {name} )"

    def is_valid(self) -> bool:
        """Check if the cached data is still valid in Ray's object store."""
        import ray
        import ray.exceptions
        try:
            for ref in self._object_refs:
                ray.wait([ref], timeout=0, fetch_local=False)
            return True
        except (ray.exceptions.ObjectLostError, ray.exceptions.RayError):
            return False

    def invalidate(self) -> None:
        """Invalidate the cached data, clearing local references."""
        self._cached_table = None

class CachedDataSet(DataSet):
    """
    A dataset backed by Ray object references stored in memory.
    """

    __slots__ = ("_object_refs", "_cached_tables", "_cache_id")

    def __init__(self, object_refs: List[ray.ObjectRef], cache_id: str) -> None:
        super().__init__([])
        self._object_refs = object_refs
        self._cached_tables: Optional[List[arrow.Table]] = None
        self._cache_id = cache_id

    def __str__(self) -> str:
        return f"CachedDataSet(cache_id={self._cache_id}, num_refs={len(self._object_refs)})"

    __repr__ = __str__

    @property
    def object_refs(self) -> List[ray.ObjectRef]:
        return self._object_refs

    @property
    def cache_id(self) -> str:
        return self._cache_id

    def _materialize(self) -> List[arrow.Table]:
        """Materialize the cached data from Ray object store."""
        if self._cached_tables is None:
            datasets = ray.get(self._object_refs)
            self._cached_tables = [ds.to_arrow_table() for ds in datasets]
        return self._cached_tables

    @property
    def num_rows(self) -> int:
        tables = self._materialize()
        return sum(t.num_rows for t in tables)

    @property
    def empty(self) -> bool:
        return len(self._object_refs) == 0

    def to_arrow_table(
        self,
        max_workers: int = 16,
        filesystem=None,
        conn=None,
    ) -> arrow.Table:
        tables = self._materialize()
        if not tables:
            return arrow.table({})
        return arrow.concat_tables(tables)

    def to_batch_reader(
        self,
        batch_size: int = 122880,
        filesystem=None,
        conn=None,
    ) -> arrow.RecordBatchReader:
        return self.to_arrow_table().to_reader(batch_size)

    def sql_query_fragment(self, filesystem=None, conn=None) -> str:
        raise NotImplementedError("CachedDataSet does not support SQL queries directly")

    def is_valid(self) -> bool:
        """Check if the cached data is still valid in Ray's object store."""
        try:
            for ref in self._object_refs:
                ray.wait([ref], timeout=0, fetch_local=False)
            return True
        except (ray.exceptions.ObjectLostError, ray.exceptions.RayError):
            return False

@dataclass
class CacheStats:
    """Statistics for DataFrame caching operations."""
    cache_count: int = 0
    cache_hit_count: int = 0
    cache_miss_count: int = 0
    unpersist_count: int = 0
    total_cached_partitions: int = 0
    active_caches: Dict[str, int] = field(default_factory=dict)

    def record_cache(self, cache_id: str, num_partitions: int) -> None:
        """Record a cache creation."""
        self.cache_count += 1
        self.total_cached_partitions += num_partitions
        self.active_caches[cache_id] = num_partitions

    def record_hit(self) -> None:
        """Record a cache hit."""
        self.cache_hit_count += 1

    def record_miss(self) -> None:
        """Record a cache miss."""
        self.cache_miss_count += 1

    def record_unpersist(self, cache_id: str) -> None:
        """Record a cache unpersist."""
        self.unpersist_count += 1
        if cache_id in self.active_caches:
            del self.active_caches[cache_id]

    def __str__(self) -> str:
        return (
            f"CacheStats(caches={self.cache_count}, hits={self.cache_hit_count}, "
            f"misses={self.cache_miss_count}, unpersists={self.unpersist_count}, "
            f"active_caches={len(self.active_caches)}, total_partitions={self.total_cached_partitions})"
        )
