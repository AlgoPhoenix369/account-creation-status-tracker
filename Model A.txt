class TestLenientReading:
    """Tests for the lenient history file reading mode."""

    def test_lenient_skips_corrupted_keeps_valid(self, temp_history_file, sample_entry):
        """Should skip corrupted entries and keep valid ones."""
        valid_json = sample_entry.model_dump_json()
        temp_history_file.write_text(
            f"{valid_json}\n"
            f"not valid json\n"
            f'{{"query": "missing response field"}}\n'
            f"{valid_json}\n"
            f"another bad line\n"
            f"{valid_json}\n"
        )
        history = CommandHistory(history_path=temp_history_file)

        entries = history._read_history_file_lenient()

        assert len(entries) == 3
        assert all(e.query == "list files" for e in entries)

    def test_lenient_returns_empty_for_all_corrupted(self, temp_history_file):
        """Should return empty list when all entries are corrupted."""
        temp_history_file.write_text(
            "bad line 1\n"
            "bad line 2\n"
            '{"incomplete": true}\n'
        )
        history = CommandHistory(history_path=temp_history_file)

        entries = history._read_history_file_lenient()

        assert entries == []

    def test_lenient_handles_empty_file(self, temp_history_file):
        """Should return empty list for empty file."""
        temp_history_file.write_text("")
        history = CommandHistory(history_path=temp_history_file)

        entries = history._read_history_file_lenient()

        assert entries == []

    def test_lenient_handles_permission_error(self, temp_history_file, sample_entry):
        """Should return empty list on permission error."""
        temp_history_file.write_text(sample_entry.model_dump_json() + "\n")
        history = CommandHistory(history_path=temp_history_file)
        os.chmod(temp_history_file, 0o000)

        try:
            entries = history._read_history_file_lenient()
            assert entries == []
        finally:
            os.chmod(temp_history_file, stat.S_IRWXU)

    def test_lenient_handles_encoding_errors(self, temp_history_file):
        """Should handle encoding errors gracefully."""
        temp_history_file.write_bytes(b'\xff\xfe invalid bytes\n')
        history = CommandHistory(history_path=temp_history_file)

        entries = history._read_history_file_lenient()

        assert entries == []
