    def test_lenient_skips_corrupted_keeps_valid(self, temp_history_file, sample_entry):
        """Lenient parsing should skip corrupted entries and keep valid ones."""

        content = "\n".join([
            valid_json,
            "not json at all",
            '{"query": "missing response field"}',

        ])
        temp_history_file.write_text(content + "\n")

        history = CommandHistory.__new__(CommandHistory)
        history.path = temp_history_file
        history.encoding = "utf-8"

        entries = history._read_history_file_lenient()
        assert len(entries) == 3
        assert all(e.query == "list files" for e in entries)

    def test_lenient_returns_empty_for_all_corrupted(self, temp_history_file):
        """Lenient parsing should return empty list if all entries are corrupted."""
        temp_history_file.write_text("bad line 1\nbad line 2\n{invalid json}\n")

        history = CommandHistory.__new__(CommandHistory)
        history.path = temp_history_file
        history.encoding = "utf-8"

        entries = history._read_history_file_lenient()
        assert entries == []

    def test_lenient_handles_permission_error(self, temp_history_file, sample_entry):
        """Lenient parsing should return empty list on permission error."""
        temp_history_file.write_text(sample_entry.model_dump_json() + "\n")
        os.chmod(temp_history_file, 0o000)

        try:
            history = CommandHistory.__new__(CommandHistory)
            history.path = temp_history_file
            history.encoding = "utf-8"

            entries = history._read_history_file_lenient()
            assert entries == []
        finally:
            os.chmod(temp_history_file, stat.S_IRWXU)

    def test_lenient_handles_encoding_errors(self, temp_history_file, sample_entry):
        """Lenient parsing should handle encoding errors with replacement."""
        valid_json = sample_entry.model_dump_json()
        temp_history_file.write_bytes(
            valid_json.encode("utf-8") + b"\n\xff\xfe bad encoding\n" + valid_json.encode("utf-8") + b"\n"
        )

        history = CommandHistory.__new__(CommandHistory)
        history.path = temp_history_file
        history.encoding = "utf-8"

        entries = history._read_history_file_lenient()
        assert len(entries) == 2
