class CachedDataSet(DataSet):
    """
    A dataset backed by Ray object references stored in memory.

    This dataset stores references to data that has been computed and cached
    in Ray's distributed object store, allowing efficient reuse across
    multiple operations without recomputation.
    """

    __slots__ = ("_object_refs", "_cached_tables", "_cache_id")

    def __init__(self, object_refs: List, cache_id: str) -> None:
        """
        Create a CachedDataSet from Ray object references.

        Parameters
        ----------
        object_refs
            List of Ray ObjectRefs pointing to computed DataSet objects.
        cache_id
            Unique identifier for this cache entry.
        """
        super().__init__([])
        self._object_refs = object_refs
        self._cached_tables: Optional[List[arrow.Table]] = None
        self._cache_id = cache_id

    def __str__(self) -> str:
        return f"CachedDataSet(cache_id={self._cache_id}, num_refs={len(self._object_refs)})"

    __repr__ = __str__

    @property
    def object_refs(self) -> List:
        """Ray ObjectRefs pointing to the cached data."""
        return self._object_refs

    @property
    def cache_id(self) -> str:
        """Unique identifier for this cache entry."""
        return self._cache_id

    def _materialize(self) -> List[arrow.Table]:
        """Materialize the cached data from Ray object store."""
        if self._cached_tables is None:
            import ray
            datasets = ray.get(self._object_refs)
            self._cached_tables = [ds.to_arrow_table() for ds in datasets]
        return self._cached_tables

    @property
    def num_rows(self) -> int:
        tables = self._materialize()
        return sum(t.num_rows for t in tables)

    @property
    def empty(self) -> bool:
        return len(self._object_refs) == 0

    def to_arrow_table(
        self,
        max_workers: int = 16,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> arrow.Table:
        tables = self._materialize()
        if not tables:
            return arrow.table({})
        return arrow.concat_tables(tables)

    def to_batch_reader(
        self,
        batch_size: int = DEFAULT_BATCH_SIZE,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> arrow.RecordBatchReader:
        return self.to_arrow_table().to_reader(batch_size)

    def sql_query_fragment(
        self,
        filesystem: fsspec.AbstractFileSystem = None,
        conn: duckdb.DuckDBPyConnection = None,
    ) -> str:
        """
        Register the cached table with DuckDB and return a query fragment.

        This follows the pattern used by ArrowTableDataSet.sql_query_fragment().
        """
        table = self.to_arrow_table()
        name = f"cached_table_{self._cache_id}"
        table.to_pandas().to_sql(name, conn, index=False)
        return f"( select * from {name} )"

    def is_valid(self) -> bool:
        """Check if the cached data is still valid in Ray's object store."""
        try:
            import ray
            import ray.exceptions
            for ref in self._object_refs:
                ray.wait([ref], timeout=0, fetch_local=False)
            return True
        except (ray.exceptions.ObjectLostError, ray.exceptions.RayError):
            return False

class CachedDataSet(DataSet):
    """
    A dataset backed by Ray object references stored in memory.
    """

    __slots__ = ("_object_refs", "_cached_tables", "_cache_id")

    def __init__(self, object_refs: List[ray.ObjectRef], cache_id: str) -> None:
        super().__init__([])
        self._object_refs = object_refs
        self._cached_tables: Optional[List[arrow.Table]] = None
        self._cache_id = cache_id

    def __str__(self) -> str:
        return f"CachedDataSet(cache_id={self._cache_id}, num_refs={len(self._object_refs)})"

    __repr__ = __str__

    @property
    def object_refs(self) -> List[ray.ObjectRef]:
        return self._object_refs

    @property
    def cache_id(self) -> str:
        return self._cache_id

    def _materialize(self) -> List[arrow.Table]:
        """Materialize the cached data from Ray object store."""
        if self._cached_tables is None:
            datasets = ray.get(self._object_refs)
            self._cached_tables = [ds.to_arrow_table() for ds in datasets]
        return self._cached_tables

    @property
    def num_rows(self) -> int:
        tables = self._materialize()
        return sum(t.num_rows for t in tables)

    @property
    def empty(self) -> bool:
        return len(self._object_refs) == 0

    def to_arrow_table(
        self,
        max_workers: int = 16,
        filesystem=None,
        conn=None,
    ) -> arrow.Table:
        tables = self._materialize()
        if not tables:
            return arrow.table({})
        return arrow.concat_tables(tables)

    def to_batch_reader(
        self,
        batch_size: int = 122880,
        filesystem=None,
        conn=None,
    ) -> arrow.RecordBatchReader:
        return self.to_arrow_table().to_reader(batch_size)

    def sql_query_fragment(self, filesystem=None, conn=None) -> str:
        raise NotImplementedError("CachedDataSet does not support SQL queries directly")

    def is_valid(self) -> bool:
        """Check if the cached data is still valid in Ray's object store."""
        try:
            for ref in self._object_refs:
                ray.wait([ref], timeout=0, fetch_local=False)
            return True
        except (ray.exceptions.ObjectLostError, ray.exceptions.RayError):
            return False

        cached_node = CachedDataSourceNode(self.session._ctx, cached_dataset)

        # Create and return a new DataFrame backed by the cached data
        cached_df = DataFrame(self.session, cached_node, recompute=False)

        # Create pre-computed tasks for the cached node so they can be reused
        npartitions = len(object_refs)
        cached_tasks = [
            CachedTask(
                self.session._runtime_ctx,
                object_ref,
                partition_index,
                npartitions,
                cache_id,
            )
            for partition_index, object_ref in enumerate(object_refs)
        ]

        # Register the cached tasks in the session's task cache
        self.session._node_to_tasks[cached_node] = cached_tasks

        # Update cache statistics
        self.session._cache_stats.cache_count += 1
        self.session._cache_stats.total_cached_partitions += npartitions
        self.session._cache_stats.active_caches[cache_id] = npartitions

        logger.info(f"Cached DataFrame with cache_id={cache_id}, {npartitions} partitions")

        return cached_df

    def is_cached(self) -> bool:
        """
        Check if this DataFrame is backed by cached data.
        """
        return isinstance(self.plan, CachedDataSourceNode)

    def unpersist(self) -> None:
        """
        Remove the cached data from memory.

        After calling this method, the cached DataFrame should not be used
        for new operations. This is useful to free up memory in Ray's object store.

        Note: Calling unpersist() multiple times is safe and will be ignored
        after the first call.
        """
        if not self.is_cached():
            logger.warning("unpersist() called on a non-cached DataFrame, ignoring")
            return

        cache_id = self.plan.cached_dataset.cache_id

        # Check if already unpersisted
        if cache_id not in self.session._cache_stats.active_caches:
            logger.debug(f"Cache {cache_id} already unpersisted, ignoring")
            return

        # Remove from session's task cache
        if self.plan in self.session._node_to_tasks:
            del self.session._node_to_tasks[self.plan]

        # Update cache statistics
        npartitions = self.session._cache_stats.active_caches.pop(cache_id, 0)
        self.session._cache_stats.total_cached_partitions -= npartitions
        self.session._cache_stats.unpersist_count += 1

        logger.info(f"Unpersisted cached DataFrame with cache_id={cache_id}, freed {npartitions} partitions")
