"""Command history management for zev.

This module provides persistent storage and retrieval of command queries and their
LLM-generated responses. History is stored as JSONL (JSON Lines) format in the user's
home directory.
"""

import json
import shutil
from datetime import datetime
from pathlib import Path
from typing import Optional

import questionary
from pydantic import BaseModel, ValidationError
from rich import print as rprint

from zev.command_selector import show_options
from zev.constants import HISTORY_FILE_NAME
from zev.llms.types import OptionsResponse


class HistoryError(Exception):
    """Base exception for history-related errors."""

    pass


class HistoryCorruptedError(HistoryError):
    """Raised when the history file contains invalid or corrupted data."""

    pass


class HistoryIOError(HistoryError):
    """Raised when file I/O operations fail."""

    pass


class CommandHistoryEntry(BaseModel):
    """A single entry in the command history.

    Attributes:
        query: The user's original query string.
        response: The LLM's response containing command suggestions.
    """

    query: str
    response: OptionsResponse


class CommandHistory:
    """Manages persistent storage of command history.

    The history file is stored as JSONL (one JSON object per line) in the user's
    home directory. The class handles file corruption gracefully by offering to
    backup and recreate the history file.

    Attributes:
        path: Path to the history file.
        max_entries: Maximum number of entries to retain (oldest are trimmed).
        encoding: Character encoding for file operations.
    """

    def __init__(self, history_path: Optional[Path] = None) -> None:
        """Initialize the command history manager.

        Args:
            history_path: Optional custom path for the history file.
                         Defaults to ~/.zevhistory.

        Raises:
            HistoryIOError: If the history file cannot be created due to
                           permission errors or disk issues.
        """
        self.path = history_path or Path.home() / HISTORY_FILE_NAME
        self.max_entries = 100
        self.encoding = "utf-8"
        self._ensure_history_file_exists()

    def _ensure_history_file_exists(self) -> None:
        """Create the history file if it doesn't exist.

        Raises:
            HistoryIOError: If the file cannot be created.
        """
        try:
            self.path.touch(exist_ok=True)
        except PermissionError as e:
            raise HistoryIOError(
                f"Permission denied creating history file at {self.path}: {e}"
            ) from e
        except OSError as e:
            raise HistoryIOError(
                f"Failed to create history file at {self.path}: {e}"
            ) from e

    def save_options(self, query: str, options: OptionsResponse) -> bool:
        """Save a query and its response to the history file.

        Args:
            query: The user's query string. Must be non-empty after stripping.
            options: The LLM response containing command suggestions.

        Returns:
            True if the entry was saved successfully, False otherwise.

        Note:
            This method logs errors via Rich but does not raise exceptions,
            allowing the application to continue even if history saving fails.
        """
        if not self._validate_entry(query, options):
            return False

        try:
            entry = CommandHistoryEntry(query=query, response=options)
            self._write_to_history_file(entry)
            return True
        except ValidationError as e:
            rprint(f"[yellow]Warning: Could not save to history (invalid data): {e}[/yellow]")
            return False
        except HistoryIOError as e:
            rprint(f"[yellow]Warning: Could not save to history: {e}[/yellow]")
            return False

    def _validate_entry(self, query: str, options: OptionsResponse) -> bool:
        """Validate a history entry before saving.

        Args:
            query: The user's query string.
            options: The LLM response.

        Returns:
            True if the entry is valid, False otherwise.
        """
        if not isinstance(query, str) or not query.strip():
            rprint("[yellow]Warning: Cannot save empty query to history[/yellow]")
            return False

        if not isinstance(options, OptionsResponse):
            rprint("[yellow]Warning: Cannot save invalid options to history[/yellow]")
            return False

        return True

    def get_history(self) -> list[CommandHistoryEntry]:
        """Retrieve all entries from the history file.

        Returns:
            A list of CommandHistoryEntry objects, empty if the history file
            is empty or unreadable. Entries are returned in chronological order
            (oldest first).

        Note:
            If the history file is corrupted, this method will prompt the user
            to backup and recreate the file.
        """
        try:
            return self._read_history_file()
        except HistoryCorruptedError:
            if self._handle_corrupted_file():
                return []
            return self._read_history_file_lenient()
        except HistoryIOError as e:
            rprint(f"[yellow]Warning: Could not read history: {e}[/yellow]")
            return []

    def _read_history_file(self) -> list[CommandHistoryEntry]:
        """Read and parse the history file strictly.

        Returns:
            List of valid history entries.

        Raises:
            HistoryCorruptedError: If any line contains invalid JSON or data.
            HistoryIOError: If the file cannot be read.
        """
        try:
            with open(self.path, "r", encoding=self.encoding) as f:
                entries = []
                for line_num, line in enumerate(f, start=1):
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        entry = CommandHistoryEntry.model_validate_json(line)
                        entries.append(entry)
                    except (json.JSONDecodeError, ValidationError) as e:
                        raise HistoryCorruptedError(
                            f"Invalid entry at line {line_num}: {e}"
                        ) from e
                return entries
        except FileNotFoundError:
            return []
        except PermissionError as e:
            raise HistoryIOError(f"Permission denied reading {self.path}: {e}") from e
        except UnicodeDecodeError as e:
            raise HistoryCorruptedError(f"Encoding error in history file: {e}") from e
        except OSError as e:
            raise HistoryIOError(f"Failed to read history file: {e}") from e

    def _read_history_file_lenient(self) -> list[CommandHistoryEntry]:
        """Read the history file, skipping invalid entries.

        Returns:
            List of valid entries, with corrupted entries silently skipped.
        """
        entries = []
        try:
            with open(self.path, "r", encoding=self.encoding, errors="replace") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        entry = CommandHistoryEntry.model_validate_json(line)
                        entries.append(entry)
                    except (json.JSONDecodeError, ValidationError):
                        continue
        except (OSError, PermissionError):
            pass
        return entries

    def _handle_corrupted_file(self) -> bool:
        """Handle a corrupted history file by offering backup and recreation.

        Returns:
            True if the user chose to backup and recreate, False to continue
            with lenient parsing.
        """
        rprint("[yellow]Warning: History file appears to be corrupted.[/yellow]")

        try:
            choice = questionary.confirm(
                "Would you like to backup and recreate the history file?"
            ).ask()
        except (KeyboardInterrupt, EOFError):
            return False

        if choice is None:
            return False

        if choice:
            return self._backup_and_recreate()
        return False

    def _backup_and_recreate(self) -> bool:
        """Backup the corrupted file and create a fresh history file.

        Returns:
            True if successful, False otherwise.
        """
        backup_path = self.path.with_suffix(
            f".backup.{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        )
        try:
            shutil.copy2(self.path, backup_path)
            rprint(f"[green]✓[/green] Backed up to {backup_path}")

            self.path.write_text("", encoding=self.encoding)
            rprint("[green]✓[/green] Created fresh history file")
            return True
        except PermissionError as e:
            rprint(f"[red]Could not backup history file (permission denied): {e}[/red]")
            return False
        except OSError as e:
            rprint(f"[red]Could not backup history file: {e}[/red]")
            return False

    def _write_to_history_file(self, new_entry: CommandHistoryEntry) -> None:
        """Append an entry to the history file and trim if necessary.

        Args:
            new_entry: The entry to append.

        Raises:
            HistoryIOError: If the file cannot be written.
        """
        try:
            with open(self.path, "a", encoding=self.encoding) as f:
                f.write(new_entry.model_dump_json() + "\n")
        except PermissionError as e:
            raise HistoryIOError(f"Permission denied writing to {self.path}: {e}") from e
        except OSError as e:
            raise HistoryIOError(f"Failed to write to history file: {e}") from e

        self._trim_history_if_needed()

    def _trim_history_if_needed(self) -> None:
        """Remove oldest entries if the history exceeds max_entries."""
        try:
            with open(self.path, "r", encoding=self.encoding) as f:
                lines = f.readlines()

            if len(lines) > self.max_entries:
                with open(self.path, "w", encoding=self.encoding) as f:
                    f.writelines(lines[-self.max_entries:])
        except (OSError, PermissionError) as e:
            rprint(f"[yellow]Warning: Could not trim history file: {e}[/yellow]")

    def display_history_options(
        self, reverse_history_entries: list[CommandHistoryEntry], show_limit: int = 5
    ) -> Optional[CommandHistoryEntry]:
        """Display an interactive menu for selecting from history entries.

        Args:
            reverse_history_entries: History entries in reverse chronological order.
            show_limit: Number of entries to show initially before "Show more...".

        Returns:
            The selected CommandHistoryEntry, None if cancelled or no selection,
            or "Cancel" string if explicitly cancelled.
        """
        if not reverse_history_entries:
            rprint("[yellow]No command history found[/yellow]")
            return None

        style = questionary.Style(
            [
                ("answer", "fg:#61afef"),
                ("question", "bold"),
                ("instruction", "fg:#98c379"),
            ]
        )

        query_options = [
            questionary.Choice(entry.query, value=entry)
            for entry in reverse_history_entries[:show_limit]
        ]

        if len(reverse_history_entries) > show_limit:
            query_options.append(questionary.Choice("Show more...", value="show_more"))

        query_options.append(questionary.Separator())
        query_options.append(questionary.Choice("Cancel"))

        try:
            selected = questionary.select(
                "Select from history:",
                choices=query_options,
                use_shortcuts=True,
                style=style,
            ).ask()
        except (KeyboardInterrupt, EOFError):
            return None

        if selected == "show_more":
            all_options = [
                questionary.Choice(entry.query, value=entry)
                for entry in reverse_history_entries
            ]
            all_options.append(questionary.Separator())
            all_options.append(questionary.Choice("Cancel"))

            try:
                return questionary.select(
                    "Select from history (showing all items):",
                    choices=all_options,
                    use_shortcuts=True,
                    style=style,
                ).ask()
            except (KeyboardInterrupt, EOFError):
                return None

        return selected

    def show_history(self) -> None:
        """Display the command history and allow the user to select and reuse a query.

        This is the main entry point for the --recent/-r command line option.
        Shows history entries in reverse chronological order (most recent first).
        """
        history_entries = self.get_history()
        if not history_entries:
            rprint("[yellow]No command history found[/yellow]")
            return

        selected_entry = self.display_history_options(list(reversed(history_entries)))

        if selected_entry in (None, "Cancel"):
            return

        commands = selected_entry.response.commands

        if not commands:
            rprint("[yellow]No commands available[/yellow]")
            return

        show_options(commands)
