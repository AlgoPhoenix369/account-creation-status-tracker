import json
import shutil
from datetime import datetime
from pathlib import Path
from typing import Optional

import questionary
from pydantic import BaseModel, ValidationError
from rich import print as rprint

from zev.command_selector import show_options
from zev.constants import HISTORY_FILE_NAME
from zev.llms.types import OptionsResponse


class HistoryError(Exception):
    """Base exception for command history operations."""

    pass


class HistoryCorruptedError(HistoryError):
    """Raised when the history file contains invalid data."""

    pass


class HistoryPermissionError(HistoryError):
    """Raised when there are permission issues with the history file."""

    pass


class CommandHistoryEntry(BaseModel):
    query: str
    response: OptionsResponse


class CommandHistory:
    """Manages persistent storage of command history in JSONL format.

    The history file is stored at ~/.zevhistory and contains one JSON object
    per line. Each entry records a user query and the LLM's response.

    Attributes:
        path: Path to the history file.
        max_entries: Maximum number of entries to retain (oldest are trimmed).
        encoding: Character encoding for file operations.
    """

    def __init__(self, path: Optional[Path] = None) -> None:
        """Initialize CommandHistory with optional custom path.

        Args:
            path: Custom path for history file. Defaults to ~/.zevhistory.

        Raises:
            HistoryPermissionError: If the history file cannot be created due to permissions.
        """
        self.path = path or (Path.home() / HISTORY_FILE_NAME)
        self.max_entries = 100
        self.encoding = "utf-8"
        self._ensure_file_exists()

    def _ensure_file_exists(self) -> None:
        """Create the history file if it doesn't exist."""
        try:
            self.path.touch(exist_ok=True)
        except PermissionError as e:
            raise HistoryPermissionError(
                f"Cannot create history file at {self.path}: permission denied"
            ) from e
        except OSError as e:
            raise HistoryError(f"Cannot create history file at {self.path}: {e}") from e

    def save_options(self, query: str, options: OptionsResponse) -> bool:
        """Save a query and its response to history.

        Args:
            query: The user's original query string.
            options: The LLM response containing command options.

        Returns:
            True if save was successful, False if it failed gracefully.

        Raises:
            ValueError: If query is empty or contains only whitespace.
        """
        if not self._validate_query(query):
            raise ValueError("Query must be a non-empty string")

        if not self._validate_options(options):
            rprint("[yellow]Warning: Invalid options response, skipping history save[/yellow]")
            return False

        entry = CommandHistoryEntry(query=query, response=options)
        return self._write_to_history_file(entry)

    def _validate_query(self, query: str) -> bool:
        """Validate that query is a non-empty string."""
        return isinstance(query, str) and len(query.strip()) > 0

    def _validate_options(self, options: OptionsResponse) -> bool:
        """Validate that options is a valid OptionsResponse."""
        if options is None:
            return False
        if not isinstance(options, OptionsResponse):
            return False
        return True

    def get_history(self) -> list[CommandHistoryEntry]:
        """Read all entries from the history file.

        Returns:
            List of CommandHistoryEntry objects, empty list if file is empty
            or cannot be read.

        Note:
            If the history file is corrupted, this method will attempt to
            offer recovery options to the user.
        """
        try:
            return self._read_history_file()
        except HistoryCorruptedError:
            if self._offer_recovery():
                return []
            return []
        except HistoryPermissionError as e:
            rprint(f"[red]Cannot read history: {e}[/red]")
            return []
        except HistoryError as e:
            rprint(f"[red]Error reading history: {e}[/red]")
            return []

    def _read_history_file(self) -> list[CommandHistoryEntry]:
        """Read and parse the history file.

        Returns:
            List of valid CommandHistoryEntry objects.

        Raises:
            HistoryCorruptedError: If the file contains invalid JSON or entries.
            HistoryPermissionError: If the file cannot be read due to permissions.
            HistoryError: For other file operation errors.
        """
        try:
            with open(self.path, "r", encoding=self.encoding) as f:
                content = f.read()
        except PermissionError as e:
            raise HistoryPermissionError(
                f"Cannot read history file: permission denied"
            ) from e
        except UnicodeDecodeError as e:
            raise HistoryCorruptedError(
                f"History file has encoding issues: {e}"
            ) from e
        except OSError as e:
            raise HistoryError(f"Cannot read history file: {e}") from e

        if not content.strip():
            return []

        entries = []
        lines = content.splitlines()

        for line_num, line in enumerate(lines, start=1):
            if not line.strip():
                continue

            try:
                entry = CommandHistoryEntry.model_validate_json(line)
                entries.append(entry)
            except json.JSONDecodeError as e:
                raise HistoryCorruptedError(
                    f"Invalid JSON at line {line_num}: {e}"
                ) from e
            except ValidationError as e:
                raise HistoryCorruptedError(
                    f"Invalid entry format at line {line_num}: {e}"
                ) from e

        return entries

    def _write_to_history_file(self, new_entry: CommandHistoryEntry) -> bool:
        """Append an entry to the history file and trim if needed.

        Args:
            new_entry: The entry to write.

        Returns:
            True if write was successful, False otherwise.
        """
        try:
            with open(self.path, "a", encoding=self.encoding) as f:
                f.write(new_entry.model_dump_json() + "\n")
        except PermissionError:
            rprint("[red]Cannot save to history: permission denied[/red]")
            return False
        except OSError as e:
            rprint(f"[red]Cannot save to history: {e}[/red]")
            return False

        self._trim_if_needed()
        return True

    def _trim_if_needed(self) -> None:
        """Remove oldest entries if file exceeds max_entries."""
        try:
            with open(self.path, "r", encoding=self.encoding) as f:
                lines = f.readlines()

            if len(lines) > self.max_entries:
                with open(self.path, "w", encoding=self.encoding) as f:
                    f.writelines(lines[-self.max_entries:])
        except (PermissionError, OSError):
            pass

    def _offer_recovery(self) -> bool:
        """Offer to backup and recreate a corrupted history file.

        Returns:
            True if recovery was performed, False if user declined or recovery failed.
        """
        rprint("[yellow]Your command history file appears to be corrupted.[/yellow]")

        try:
            should_recover = questionary.confirm(
                "Would you like to backup the corrupted file and start fresh?"
            ).ask()
        except (KeyboardInterrupt, EOFError):
            return False

        if not should_recover:
            return False

        return self._perform_recovery()

    def _perform_recovery(self) -> bool:
        """Backup the corrupted file and create a fresh history file.

        Returns:
            True if recovery succeeded, False otherwise.
        """
        backup_path = self.path.with_suffix(
            f".backup.{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        )

        try:
            shutil.copy2(self.path, backup_path)
            rprint(f"[green]✓[/green] Backed up corrupted file to {backup_path}")
        except (PermissionError, OSError) as e:
            rprint(f"[red]Could not create backup: {e}[/red]")
            return False

        try:
            self.path.write_text("", encoding=self.encoding)
            rprint("[green]✓[/green] Created fresh history file")
            return True
        except (PermissionError, OSError) as e:
            rprint(f"[red]Could not recreate history file: {e}[/red]")
            return False

    def display_history_options(
        self, reverse_history_entries: list[CommandHistoryEntry], show_limit: int = 5
    ) -> Optional[CommandHistoryEntry]:
        """Display history entries for user selection.

        Args:
            reverse_history_entries: History entries in reverse chronological order.
            show_limit: Number of entries to show initially.

        Returns:
            The selected CommandHistoryEntry, or None if cancelled.
        """
        if not reverse_history_entries:
            rprint("[yellow]No command history found[/yellow]")
            return None

        style = questionary.Style(
            [
                ("answer", "fg:#61afef"),
                ("question", "bold"),
                ("instruction", "fg:#98c379"),
            ]
        )

        query_options = [
            questionary.Choice(entry.query, value=entry)
            for entry in reverse_history_entries[:show_limit]
        ]

        if len(reverse_history_entries) > show_limit:
            query_options.append(questionary.Choice("Show more...", value="show_more"))

        query_options.append(questionary.Separator())
        query_options.append(questionary.Choice("Cancel"))

        try:
            selected = questionary.select(
                "Select from history:",
                choices=query_options,
                use_shortcuts=True,
                style=style,
            ).ask()
        except (KeyboardInterrupt, EOFError):
            return None

        if selected == "show_more":
            all_options = [
                questionary.Choice(entry.query, value=entry)
                for entry in reverse_history_entries
            ]
            all_options.append(questionary.Separator())
            all_options.append(questionary.Choice("Cancel"))

            try:
                return questionary.select(
                    "Select from history (showing all items):",
                    choices=all_options,
                    use_shortcuts=True,
                    style=style,
                ).ask()
            except (KeyboardInterrupt, EOFError):
                return None

        return selected

    def show_history(self) -> None:
        """Display command history and allow user to select and execute a previous query.

        This is the main entry point for the --recent/-r command flag.
        """
        history_entries = self.get_history()
        if not history_entries:
            rprint("[yellow]No command history found[/yellow]")
            return

        selected_entry = self.display_history_options(list(reversed(history_entries)))

        if selected_entry in (None, "Cancel"):
            return

        commands = selected_entry.response.commands

        if not commands:
            rprint("[yellow]No commands available[/yellow]")
            return

        show_options(commands)
